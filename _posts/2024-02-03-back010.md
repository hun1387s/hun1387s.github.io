두가지 방법이 있다.

 

1. 재귀 호출을 이용한 방법

   ``` cpp
   int RecursiveFibonacci(int n)
   {
   	if (n == 0)
   		return 0;
   	else if (n == 1)
   		return 1;
   	else
   		return RecursiveFibonacci(n - 1) + RecursiveFibonacci(n - 2);
   }
   ```

   

2. 반복문을 이용한 방법

   ``` cpp
   int Fibonacci(int n)
   {
   	if (n == 0)
   		return 0;
   	else if (n == 1)
   		return 1;
   	else
   	{
   		int tmp1 = 0;
   		int tmp2 = 1;
   		int value = 0;
   
   		for (int i = 2; i <= n; i++) {
   			value = tmp1 + tmp2;
   
   			tmp1 = tmp2;
   			tmp2 = value;
   		}
   
   		return value;
   	}
   }
   ```
